gauge.js

Native JavaScript gauge control

  • Download Source
  • Download Minimized
  • Plugin for JQuery
  • Никаких картинок, спрайтов или CSS - только canvas
  • Никаких зависимостей. Jquery поддерживается, но совершенно не нужен
  • Гибкая система настроек
  • Любые значения шкалы в пределах до 10,000
  • Доступно и понятно стилизуем через настройки
  • Устанавливается через bower

Использование

var g = Gauge({
    id: "canvas",       // ID элемента для установки прибора
    width: 400,         // Размер
    name: "Вольтметр",  // Имя прибора
    index: "кВ",        // Единицы измерения
    scaleInside: false, // Шкала внесена внутрь
    maxValue: 6,        // Максимальное значение
    colors: {           // Цвета:
        background: '#000', // — задника
        texts: '#fff',      // — цифр, букв и разметки
        hands: '#f00',      // — стрелки
        base: '#02fefe',    // — шкалы
        warning: '#02aeee', // — стремной зоны
        danger: '#f00'      // — опасной зоны
    },
    scaleRange: {
        warning: 4.5,   // начало стремной зоны
        danger: 5.5     // начало опасной зоны
    },
    value: 4        // текущее значение
});

g.setValue(5);      // Установить новое значение в единицах
g.set(0.25);        // Установить новое значение в долях единицы

Примеры использования

Спидометр и вольтметр

Gauge({
    id: "examples",
    width: 400,
    name: 'Спидометр',
    index: "км / ч",
    scaleInside: true,
    maxValue: 250,
    colors: {
        background: '#000',
        texts: '#fff',
        hands: '#f00',
        base: '#02fefe',
        warning: '#fa0',
        danger: '#f00'
    },
    scaleRange: {
        warning: 200,
        danger: 235
    },
    value: 60
});

Gauge({
    id: "examples",
    width: 400,
    name: 'СДЕЛАНО В СССР',
    index: "V",
    maxValue: 1000
});

Поддержка браузерами

Прибор корректно отображается во всех современных браузерах*:

  • Chrome 6.0+
  • Safari 3.2+
  • Firefox 3.5+
  • IE 9
  • Opera 9.6+
  • Mobile Safari (iOS 3.0+)
  • Android 2.1+

*Чисто теоритически ;) Я не проверял
